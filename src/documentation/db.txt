clients:
    id(SERIAL PRIMARY KEY).

    name(VARCHAR).

    password(VARCHAR).

    email(VARCHAR).

    thema(BOOL)
        verdadeiro = Claro;
        falso = Escuro.

    created_at(TIMESTAMP)
    
    language(INT)
        1 = Português;
        2 = Inglês;
        3 = Francês;
        4 = Espanhol.


list:
    id(SERIAL PRIMARY KEY),

    name(VARCHAR),

    update_period(INT)
        1 = Não atualiza;
        2 = Dia a dia;
        3 = semana a semana;
        4 = mês a mês.
    
    bg_color(INT)
        1 = Sem;
        2 = Azul;
        3 = Roxo;
        4 = Laranja;
        5 = Vermelho;
        6 = Amarelo.
    
    id_clients(INT)
        FOREIGN KEY(id_clients) REFERENCES clients(id).

itens:

    id(SERIAL PRIMARY KEY).
    
    name(VARCHAR).
    
    amount(FLOAT).
    
    type (SMALLINT).

    completed (BOOLEAN).
    
    type(INT)
        1 = Unidade;
        2 = Gramas;
        3 = Kilogramas;
        4 = Metros;
        5 = Centímetros;
        6 = Pacotes.

    completed(BOOL)
        falso = Em andamento;
        verdadeiro = Concluídas.

    id_list(INT)
        FOREIGN KEY(id_list) REFERENCES list(id)


##SQL
CREATE TABLE clients (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    password TEXT NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    thema BOOLEAN DEFAULT true,
    language TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE list (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    update_period SMALLINT DEFAULT 1,
    bg_color SMALLINT DEFAULT 1,
    id_clients INT,
    FOREIGN KEY(id_clients) REFERENCES clients(id)
);

CREATE TABLE itens (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    amount FLOAT DEFAULT 1,
    type SMALLINT DEFAULT 1,
    completed BOOLEAN DEFAULT false,
    id_list INT,
    FOREIGN KEY(id_list) REFERENCES list(id)
);
